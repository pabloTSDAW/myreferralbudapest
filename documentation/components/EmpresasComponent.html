import { Component, OnInit } from '@angular/core';
import { FirebaseService } from '../firebase.service';

@Component({
  selector: 'app-empresas',
  templateUrl: './empresas.component.html',
  styleUrls: ['./empresas.component.css']
})
export class EmpresasComponent implements OnInit {

  empresas: Array<any>;

  constructor(private firebase: FirebaseService) { }

    ngOnInit(){
      this.firebase.getEmpresas().subscribe(data => this.empresas = data);
    }

    muestra(elem){
      this.firebase.setEmpresa(elem);
    }
  }

<section id="empresas">
  <div class="container p-5">
    <div class="row">
      <div class="col-lg-12 text-center mb-5">
        <h2 class="section-heading mb-4">Empresas</h2>
        <h5>Entra a formar parte de una de las grandes empresas a nivel mundial, sitas en Budapest.</h5>
      </div>
    </div>
    <div class="row">
      <div class="col-12 col-md-3 mb-3 text-center" *ngFor="let empresa of empresas">
        <div>
          <img src="../../assets/images/empresas/{{empresa.nombre}}.png" [routerLink]="['/empresa']" (click)="muestra(empresa)">
        </div>
      </div>
    </div>
  </div>
</section>
